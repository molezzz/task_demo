def monitor_configs
  {
    default_source: ->{ self.user },
    created: {
      name: :to_todo,
      filter: ->(to, from){
        #这里的self指代被创建的 comment
        !self.todo_id.nil?
      },
      belongs_to: :todo,
      source: ->{
        self.todo.event_source || self.user
      }
    }
  }
